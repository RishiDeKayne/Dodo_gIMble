#!/bin/bash
#SBATCH --job-name=pic
#SBATCH --time=0-24:00:00 # Wall clock time limit in Days-Hours:min:seconds
#SBATCH --account=co_genomicdata
#SBATCH --partition=savio3
#SBATCH --output=pic.%j.out # output file
#SBATCH --error=pic.%j.err # error file
#SBATCH --ntasks=1 # Run 1 job
#SBATCH --ntasks-per-node=1 # One task per computer
#SBATCH --cpus-per-task=24 # 2 CPUs per job

module load java/22.0.1

picard AddOrReplaceReadGroups \
       I=Dodo.bam \
       O=DodoRG.bam \
       RGID=4 \
       RGLB=lib1 \
       RGPL=ILLUMINA \
       RGPU=unit1 \
       RGSM=Dodo

picard AddOrReplaceReadGroups \
       I=Manumea.bam \
       O=ManumeaRG.bam \
       RGID=4 \
       RGLB=lib1 \
       RGPL=ILLUMINA \
       RGPU=unit1 \
       RGSM=Manumea

picard AddOrReplaceReadGroups \
       I=Solitaire.bam \
       O=SolitaireRG.bam \
       RGID=4 \
       RGLB=lib1 \
       RGPL=ILLUMINA \
       RGPU=unit1 \
       RGSM=Solitaire

picard AddOrReplaceReadGroups \
       I=SMF045.bam \
       O=SMF045RG.bam \
       RGID=4 \
       RGLB=lib1 \
       RGPL=ILLUMINA \
       RGPU=unit1 \
       RGSM=SMF045

touch all.done

##run
sbatch addRG.sh
